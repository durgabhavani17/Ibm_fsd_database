1)select e1.reportsto ,count(*) from employees e1 join employees e2 on e1.employeenumber=e2.employeenumber group by e1.reportsto;


2)select distinct p.productname,c.customername from customers c join orders o1 on c.customernumber=o1.customernumber join orderdetails o2 on o2.ordernumber=o1.ordernumber join products p
on p.productcode=o2.productcode order by c.customername asc limit 1;  

3)Select P.productname,sum(OD.priceEach) from Products P join OrderDetails OD on P.productCode=OD.productCode group by P.productname;

4)Select E.firstName,sum(P.amount) from Employees E join Employees F on E.EmployeeNumber=F.reportsTo
join Customers C on C.salesRepEmployeeNumber=F.EmployeeNumber
join Payments P on P.customerNumber=C.customerNumber group by E.firstName order by sum(P.amount) desc;

5)select extract(month from o.orderdate) as month ,count(*)  as totalorders,p.productline from customers c join orders o on c.customernumber=o.customernumber 
 join orderdetails o2 on o2.ordernumber=o.ordernumber join products p  on p.productcode=o2.productcode  group by month,p.productline ;
